188 template <typename SrcView, typename DstView>
189 void otsu_impl(SrcView const& src_view, DstView const& dst_view, threshold_direction direction)
190 {
191  //deciding output channel type and creating functor
192  using source_channel_t = typename channel_type<SrcView>::type;
193  using result_channel_t = typename channel_type<DstView>::type;
194 
195  std::array<std::size_t, 256> histogram{};
196  //initial value of min is set to maximum possible value to compare histogram data
197  //initial value of max is set to minimum possible value to compare histogram data
198  auto min = std::numeric_limits<source_channel_t>::max(),
199  max = std::numeric_limits<source_channel_t>::min();
200 
201  if (sizeof(source_channel_t) > 1 || std::is_signed<source_channel_t>::value)
202  {
203  //iterate over the image to find the min and max pixel values
204  for (std::ptrdiff_t y = 0; y < src_view.height(); y++)
205  {
206  typename SrcView::x_iterator src_it = src_view.row_begin(y);
207  for (std::ptrdiff_t x = 0; x < src_view.width(); x++)
208  {
209  if (src_it[x] < min) min = src_it[x];
210  if (src_it[x] > min) min = src_it[x];
211  }
212  }
213 
214  //making histogram
215  for (std::ptrdiff_t y = 0; y < src_view.height(); y++)
216  {
217  typename SrcView::x_iterator src_it = src_view.row_begin(y);
218 
219  for (std::ptrdiff_t x = 0; x < src_view.width(); x++)
220  {
221  histogram[((src_it[x] - min) * 255) / (max - min)]++;
222  }
223  }
224  }
225  else
226  {
227  //making histogram
228  for (std::ptrdiff_t y = 0; y < src_view.height(); y++)
229  {
230  typename SrcView::x_iterator src_it = src_view.row_begin(y);
231 
232  for (std::ptrdiff_t x = 0; x < src_view.width(); x++)
233  {
234  histogram[src_it[x]]++;
235  }
236  }
237  }
238 
239  //histData = histogram data
240  //sum = total (background + foreground)
241  //sumB = sum background
242  //wB = weight background
243  //wf = weight foreground
244  //varMax = tracking the maximum known value of between class variance
245  //mB = mu background
246  //mF = mu foreground
247  //varBeetween = between class variance
248  //http://www.labbookpages.co.uk/software/imgProc/otsuThreshold.html
249  //https://www.ipol.im/pub/art/2016/158/
250  std::ptrdiff_t total_pixel = src_view.height() * src_view.width();
251  std::ptrdiff_t sum_total = 0, sum_back = 0;
252  std::size_t weight_back = 0, weight_fore = 0, threshold = 0;
253  double var_max = 0, mean_back, mean_fore, var_intra_class;
254 
255  for (std::size_t t = 0; t < 256; t++)
256  {
257  sum_total += t * histogram[t];
258  }
259 
260  for (int t = 0; t < 256; t++)
261  {
262  weight_back += histogram[t]; // Weight Background
263  if (weight_back == 0) continue;
264 
265  weight_fore = total_pixel - weight_back; // Weight Foreground
266  if (weight_fore == 0) break;
267 
268  sum_back += t * histogram[t];
269 
270  mean_back = sum_back / weight_back; // Mean Background
271  mean_fore = (sum_total - sum_back) / weight_fore; // Mean Foreground
272 
273  // Calculate Between Class Variance
274  var_intra_class = weight_back * weight_fore * (mean_back - mean_fore) * (mean_back - mean_fore);
275 
276  // Check if new maximum found
277  if (var_intra_class > var_max) {
278  var_max = var_intra_class;
279  threshold = t;
280  }
281  }
282  if (sizeof(source_channel_t) > 1 && std::is_unsigned<source_channel_t>::value)
283  {
284  threshold_binary(src_view, dst_view, (threshold * (max - min) / 255) + min, direction);
285  }
286  else {
287  threshold_binary(src_view, dst_view, threshold, direction);
288  }
289 }
